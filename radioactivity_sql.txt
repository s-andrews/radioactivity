DROP DATABASE IF EXISTS `Radioactivity`;
CREATE DATABASE Radioactivity;
USE Radioactivity;

CREATE TABLE Isotope (
       isotope_id INT AUTO_INCREMENT PRIMARY KEY,
       element VARCHAR (2),
       mw INT,
       half_life FLOAT,
       site_holding_limit INT,
       solid_monthly_disposal_limit INT,
       liquid_monthly_disposal_limit INT,
       emission_type ENUM("alpha","beta","gamma")
       );

CREATE TABLE Building (
       building_id INT AUTO_INCREMENT PRIMARY KEY,
       number INT,
       name VARCHAR(100)
       );

CREATE TABLE Person (
       person_id INT AUTO_INCREMENT PRIMARY KEY,
       first_name VARCHAR(50),
       last_name VARCHAR(50),
       username VARCHAR(20),
       email VARCHAR(100),
       phone VARCHAR(50),
       active BOOLEAN
       );

CREATE TABLE Received (
       received_id INT AUTO_INCREMENT PRIMARY KEY,
       date DATE,
       isotope_id INT,
       person_id INT,
       input_person_id INT,
       building_id INT,
       activity FLOAT,
       product_code VARCHAR(50),
       fully_decayed DATE
       );

CREATE TABLE Transfer_disposal (
       transfer_waste_id INT AUTO_INCREMENT PRIMARY KEY,
       date DATE,
       drum_id INT,
       isotope_id INT,
       person_id INT,
       input_person_id INT,
       building_id INT,
       activity FLOAT,
       fully_decayed DATE
       );

CREATE TABLE Liquid_disposal (
       liquid_waste_id INT AUTO_INCREMENT PRIMARY KEY,
       date DATE,
       isotope_id INT,
       person_id INT,
       input_person_id INT,
       building_id INT,
       activity FLOAT,
       fully_decayed DATE
       );

CREATE TABLE Drum (
       drum_id INT AUTO_INCREMENT PRIMARY KEY,
       date_started DATE,
       date_removed DATE,
       material ENUM("solid","liquid")
       );


CREATE TABLE Audit (
       audit_id INT AUTO_INCREMENT PRIMARY KEY,
       building_id INT,
       date DATE,
       isotope_id INT,
       activity FLOAT
       );
